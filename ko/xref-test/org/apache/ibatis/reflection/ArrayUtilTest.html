16  package org.apache.ibatis.reflection;
17  
18  import static org.junit.jupiter.api.Assertions.*;
19  
20  import java.util.Arrays;
21  
22  import org.junit.jupiter.api.Test;
23  
24  class ArrayUtilTest extends ArrayUtil {
25  
26    @Test
27    void testHashCode() {
28      Object arr;
29      arr = new long[] { 1 };
30      assertEquals(Arrays.hashCode((long[]) arr), ArrayUtil.hashCode(arr));
31      arr = new int[] { 1 };
32      assertEquals(Arrays.hashCode((int[]) arr), ArrayUtil.hashCode(arr));
33      arr = new short[] { 1 };
34      assertEquals(Arrays.hashCode((short[]) arr), ArrayUtil.hashCode(arr));
35      arr = new char[] { 1 };
36      assertEquals(Arrays.hashCode((char[]) arr), ArrayUtil.hashCode(arr));
37      arr = new byte[] { 1 };
38      assertEquals(Arrays.hashCode((byte[]) arr), ArrayUtil.hashCode(arr));
39      arr = new boolean[] { true };
40      assertEquals(Arrays.hashCode((boolean[]) arr), ArrayUtil.hashCode(arr));
41      arr = new float[] { 1f };
42      assertEquals(Arrays.hashCode((float[]) arr), ArrayUtil.hashCode(arr));
43      arr = new double[] { 1d };
44      assertEquals(Arrays.hashCode((double[]) arr), ArrayUtil.hashCode(arr));
45      arr = new Object[] { "str" };
46      assertEquals(Arrays.hashCode((Object[]) arr), ArrayUtil.hashCode(arr));
47  
48      assertEquals(0, ArrayUtil.hashCode(null));
49      assertEquals("str".hashCode(), ArrayUtil.hashCode("str"));
50      assertEquals(Integer.valueOf(1).hashCode(), ArrayUtil.hashCode(1));
51    }
52  
53    @Test
54    void testequals() {
55      assertTrue(ArrayUtil.equals(new long[] { 1 }, new long[] { 1 }));
56      assertTrue(ArrayUtil.equals(new int[] { 1 }, new int[] { 1 }));
57      assertTrue(ArrayUtil.equals(new short[] { 1 }, new short[] { 1 }));
58      assertTrue(ArrayUtil.equals(new char[] { 1 }, new char[] { 1 }));
59      assertTrue(ArrayUtil.equals(new byte[] { 1 }, new byte[] { 1 }));
60      assertTrue(ArrayUtil.equals(new boolean[] { true }, new boolean[] { true }));
61      assertTrue(ArrayUtil.equals(new float[] { 1f }, new float[] { 1f }));
62      assertTrue(ArrayUtil.equals(new double[] { 1d }, new double[] { 1d }));
63      assertTrue(ArrayUtil.equals(new Object[] { "str" }, new Object[] { "str" }));
64  
65      assertFalse(ArrayUtil.equals(new long[] { 1 }, new long[] { 2 }));
66      assertFalse(ArrayUtil.equals(new int[] { 1 }, new int[] { 2 }));
67      assertFalse(ArrayUtil.equals(new short[] { 1 }, new short[] { 2 }));
68      assertFalse(ArrayUtil.equals(new char[] { 1 }, new char[] { 2 }));
69      assertFalse(ArrayUtil.equals(new byte[] { 1 }, new byte[] { 2 }));
70      assertFalse(ArrayUtil.equals(new boolean[] { true }, new boolean[] { false }));
71      assertFalse(ArrayUtil.equals(new float[] { 1f }, new float[] { 2f }));
72      assertFalse(ArrayUtil.equals(new double[] { 1d }, new double[] { 2d }));
73      assertFalse(ArrayUtil.equals(new Object[] { "str" }, new Object[] { "rts" }));
74  
75      assertTrue(ArrayUtil.equals(null, null));
76      assertFalse(ArrayUtil.equals(new long[] { 1 }, null));
77      assertFalse(ArrayUtil.equals(null, new long[] { 1 }));
78  
79      assertTrue(ArrayUtil.equals(1, 1));
80      assertTrue(ArrayUtil.equals("str", "str"));
81    }
82  
83    @Test
84    void testToString() {
85      Object arr;
86      arr = new long[] { 1 };
87      assertEquals(Arrays.toString((long[]) arr), ArrayUtil.toString(arr));
88      arr = new int[] { 1 };
89      assertEquals(Arrays.toString((int[]) arr), ArrayUtil.toString(arr));
90      arr = new short[] { 1 };
91      assertEquals(Arrays.toString((short[]) arr), ArrayUtil.toString(arr));
92      arr = new char[] { 1 };
93      assertEquals(Arrays.toString((char[]) arr), ArrayUtil.toString(arr));
94      arr = new byte[] { 1 };
95      assertEquals(Arrays.toString((byte[]) arr), ArrayUtil.toString(arr));
96      arr = new boolean[] { true };
97      assertEquals(Arrays.toString((boolean[]) arr), ArrayUtil.toString(arr));
98      arr = new float[] { 1f };
99      assertEquals(Arrays.toString((float[]) arr), ArrayUtil.toString(arr));
100     arr = new double[] { 1d };
101     assertEquals(Arrays.toString((double[]) arr), ArrayUtil.toString(arr));
102     arr = new Object[] { "str" };
103     assertEquals(Arrays.toString((Object[]) arr), ArrayUtil.toString(arr));
104 
105     assertEquals(Integer.valueOf(1).toString(), ArrayUtil.toString(1));
106     assertEquals("null", ArrayUtil.toString(null));
107   }
